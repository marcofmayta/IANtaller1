# Función para predecir tarifa
def predict_taxi_fare(distance, duration, passengers, pickup_borough, dropoff_borough):
    """
    Función para predecir la tarifa de un viaje en taxi.
    
    Parámetros:
    - distance: distancia en millas
    - duration: duración en minutos
    - passengers: número de pasajeros
    - pickup_borough: borough de recogida
    - dropoff_borough: borough de destino
    
    Retorna:
    - Predicción de tarifa
    """
    input_data = pd.DataFrame({
        'distance': [distance],
        'duration': [duration],
        'passengers': [passengers],
        'pickup_borough': [pickup_borough],
        'dropoff_borough': [dropoff_borough]
    })
    
    input_encoded = pd.get_dummies(input_data, columns=['pickup_borough', 'dropoff_borough'], drop_first=True)
    for col in X.columns:
        if col not in input_encoded.columns:
            input_encoded[col] = 0
    input_encoded = input_encoded[X.columns]
    
    input_scaled = loaded_scaler.transform(input_encoded)
    fare_prediction = loaded_model.predict(input_scaled)
    return fare_prediction[0]
